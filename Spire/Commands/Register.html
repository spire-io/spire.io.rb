def run(args)
  parse_options(args)
  spire = Spire.new(CLI.url)
  spire.register(:email => config[:email], :password => config[:password])
  CLI.rc["secret"] = spire.secret
  CLI.save_rc
  $stdout.puts spire.secret
end
