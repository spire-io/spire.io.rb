def self.run(subcommand,*args)
  if command = COMMANDS[subcommand]
    begin
      command.run(args)
    rescue => e
      $stderr.puts "spire: #{e.message}"
      exit(-1)
    end
  else
    usage "#{subcommand} is not a supported command"
  end
end

def self.usage(message)
  $stderr.puts "spire: #{message}"
  $stderr.puts "Usage: spire <subcommand> <options>\n\nThe spire command provides command line access to the spire.io API.\n\nValid commands:\n\nregister     Register a new account\nconsole      Open up an IRB session with an open spire session.\n\nYou can get more options for any command with --help or -h.\n"
  exit(-1)
end
